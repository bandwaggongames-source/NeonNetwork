        // RENDER GAMES
        function renderGames(searchTerm = '') {
            const grid = document.getElementById('gamesGrid');
            grid.innerHTML = '';

            let filtered = allGames;
            if (searchTerm) {
                filtered = allGames.filter(game => 
                    game.title.toLowerCase().includes(searchTerm.toLowerCase()) ||
                    game.category.toLowerCase().includes(searchTerm.toLowerCase())
                );
            }

            if (filtered.length === 0) {
                grid.innerHTML = '<p class="loading-message">No games found ❌</p>';
                return;
            }

            filtered.forEach(game => {
                const card = document.createElement('div');
                card.className = 'game-card';
                
                const colors = ['#ff006e', '#00bbff', '#ffd60a', '#00d9ff', '#ff006e', '#00bb41'];
                const randomColor = colors[Math.floor(Math.random() * colors.length)];
                
                const imgSvg = `data:image/svg+xml,%3Csvg xmlns='http://www.w3.org/2000/svg' viewBox='0 0 160 180'%3E%3Crect fill='${encodeURIComponent(randomColor)}' width='160' height='180'/%3E%3Ctext x='50%25' y='50%25' fill='white' text-anchor='middle' dy='.3em' font-size='40'%3E🎮%3C/text%3E%3C/svg%3E`;
                
                const img = document.createElement('img');
                img.src = imgSvg;
                img.alt = game.title;
                
                const overlay = document.createElement('div');
                overlay.className = 'game-card-overlay';
                overlay.innerHTML = `
                    <div class="game-card-title">${game.title}</div>
                    <div class="game-card-category">${game.category}</div>
                `;

                card.appendChild(img);
                card.appendChild(overlay);
                card.addEventListener('click', () => playGame(game.title, game.url));

                grid.appendChild(card);
            });
        }

        // PLAY GAME
        function playGame(title, url) {
            document.getElementById('playerTitle').textContent = title;
            document.getElementById('gameFrame').src = url;
            document.getElementById('playerModal').classList.add('active');
        }

        function closePlayer() {
            document.getElementById('playerModal').classList.remove('active');
            document.getElementById('gameFrame').src = '';
        }

        // SEARCH
        document.getElementById('searchInput').addEventListener('input', (e) => {
            renderGames(e.target.value);
        });

        // TAB SWITCHING
        document.querySelectorAll('.nav-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
                e.target.classList.add('active');
                
                if (e.target.dataset.tab === 'favorites') {
                    if (favorites.length === 0) {
                        document.getElementById('gamesGrid').innerHTML = '<p class="loading-message">No favorites added yet ⭐</p>';
                    } else {
                        renderGames('');
                    }
                } else {
                    renderGames('');
                }
            });
        });

        // CLOSE MODAL ON ESC
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') {
                closePlayer();
            }
        });

        // INITIALIZE
        setTimeout(() => {
            document.getElementById('introScreen').style.display = 'none';
            renderGames();
        }, 4000);
    </script>
